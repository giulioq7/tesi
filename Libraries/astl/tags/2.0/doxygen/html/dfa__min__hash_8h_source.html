119  template <class ForwardI>
120  bool DFA_min_hash<CharTraits>::insert(ForwardI first, ForwardI last)
121  {
122  {
123  cursor<DFA_min_hash<CharTraits> > c(*this);
124  if (match(c, first, last)) return false; // word exists ?
125  }
126  if (!super::ready) super::prepare(); // build the structure if necessary
127  stack_cursor<forward_cursor<self> > c(forwardc(*this));
128  bool duplicating;
129 
130  for(duplicating = false; first != last; ++first) {
131  if (!duplicating)
132  this->pool(this->height(c.src())).erase(c.src());
133 
134  if (!c.find(*first)) {
135  this->set_trans(c.src(), *first, super::new_state());
136  duplicating = true;
137  c.find(*first);
138  }
139  else
140  if (super::Q[c.aim()]->degree_in() > 1) {
141  this->change_trans(c.src(), *first, this->duplicate_state(c.aim()));
142  duplicating = true;
143  c.find(*first);
144  }
145  c.forward();
146  }
147 
148  if (!duplicating)
149  this->pool(this->height(c.src())).erase(c.src());
150 
151  if (!this->final(c.src())) {
152  this->tag(c.src()).wc() += 1;
153  this->final(c.src()) = true;
154  }
155 
156  unsigned int h = this->height(c.src());
157 
158  for(; c.backward(); ) {
159  // try to insert the current state in its height pool
160  // if an equivalent state is found its ID is returned, otherwise
161  // it is inserted
162  pair<typename super::container::iterator, bool> i =
163  this->pool(this->height(c.aim())).insert(c.aim());
164  if (i.second == false) { // found an equivalent state ?
165  state_type tmp = c.aim();
166  // redirect the transition towards equivalent state
167  this->change_trans(c.src(), c.letter(), *i.first);
168  this->tag(*i.first).wc() = this->tag(tmp).wc();
169  this->del_state(tmp);
170  }
171  this->tag(c.src()).wc() += 1; // update hashing
172  h = max(this->height(c.src()), h+1); //Precond: states height initialized with 0
173  this->height(c.src(), h);
174  }
175  this->pool(this->height(super::initial())).insert(super::initial());
176  ++super::wc_;
177  return true;
178  }

180 // This specialization of the ordering of states takes advantage of hash tags:
181 template <typename Transition>
182 inline
183 bool operator<(const state<Transition, hash_tag> &x, const state<Transition, hash_tag> &y) {
184  if (x.tag().wc() < y.tag().wc()) return true;
185  if (x.tag().wc() == y.tag().wc()) {
186  if (x.final() < y.final()) return true;
187  if (x.final() == y.final()) {
188  if (x.size() < y.size()) return true;
189  else
190  if (x.size() == y.size()) {
191  typename state<Transition, hash_tag>::const_iterator i = x.begin(), j = x.end(), k = y.begin();
192  while (i != j)
193  if (*i < *k) return true;
194  else if (*k < *i) return false;
195  else { ++i; ++k; }
196  return false; // equivalent, not lower than
197  }
198  }
199  }
200  return false;
201 }
