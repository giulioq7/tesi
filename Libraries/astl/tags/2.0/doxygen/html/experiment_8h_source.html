523  inline
524  void non_deterministic_regexp_capture_cursor::preprocess_repeats() {
525  vector<node> &v = *e;
526  for(vector<node>::size_type p = 0; p < v.size();) {
527  if (v[p].type != REPEAT) ++p;
528  else {
529  // look for what is repeated: letter, (exp), ?, +, *, ANY
530  vector<node>::size_type first = p - 1, last = p;
531  for(; first > 0; --first) {
532  if (v[first].type == LETTER || v[first].type == CLOSE || v[first].type == ANY)
533  break;
534  }
535  if (v[first].type == CLOSE) {
536  int count = 1;
537  while(count != 0 && --first > 0)
538  if (v[first].type == CLOSE) ++count;
539  else if (v[first].type == OPEN) --count;
540  }
541  // the range [first, last) contains the pattern to be repeated
542  // enclose it in parenthesis because of such things: a{0,2}{3}
543  // which would otherwise be rewritten: a?a?{3} but shoud be written: (a?a?){3}
544  // limiting to 128
545  short n = (v[p].n > 128 ? 128 : v[p].n),
546  m = (v[p].m > 128 ? 128 : v[p].m);
547  v.insert(v.begin() + first, node(OPEN));
548  ++p;
549  ++last;
550  ++first;
551  // remove repeat node
552  if (n == 0)
553  v[p] = node(QUESTION);
554  else
555  v.erase(v.begin() + p);
556  vector<node> tmp;
557  if (m == -1)
558  tmp.reserve((last + 1 - first) * n + 1);
559  else
560  tmp.reserve((last + 1 - first) * m + m - n + 1);
561 
562  while(n > 0 || m > 0) {
563  --n;
564  --m;
565  if (n <= 0) { // end of mandatory repeats ?
566  if (m < n) { // case {n,}
567  tmp.push_back(node(PLUS));
568  continue; // end of processing
569  }
570  if (m == 0)
571  continue; // end of processing
572  }
573  tmp.push_back(node(CONCAT));
574  for(vector<node>::size_type tmp_p = first; tmp_p < last; ++tmp_p)
575  tmp.push_back(v[tmp_p]);
576  if (n <= 0)
577  tmp.push_back(node(QUESTION));
578  }
579  tmp.push_back(node(CLOSE));
580  v.insert(v.begin() + p, tmp.begin(), tmp.end());
581  p += tmp.size();
582  }
583  }
584  }

762  // read char at position first and advance first
763  // may return an empty char when encountering \E
764  inline
765  int non_deterministic_regexp_capture_cursor::next_char(bitset<256>& letter, const char *begin, const char*& first,
766  const char *last, bool& literal,
767  const map<string, bitset<256> >& char_sets, bool icase) {
768  letter.reset();
769 
770  // anchores
771  if ((*first == '$' || *first == '^') && !literal) {
772  letter.set((unsigned char) 0); // beginning/end of line is '\0'
773  ++first;
774  return -1;
775  }
776 
777  if (*first == '\\') {
778  if (++first == last)
779  return token_error(first - begin, "unterminated escaped sequence");
780 
781  // quote sequence \Q ... \E
782  if (*first == 'Q') {
783  if (literal) {
784  letter.set((unsigned char) '\\');
785  return -1;
786  }
787  else {
788  literal = true;
789  if (++first == last)
790  return token_error(first - begin, "unterminated quote sequence");
791  return next_char(letter, begin, first, last, literal, char_sets, icase);
792  }
793  }
794  if (*first == 'E') {
795  if (!literal) {
796  letter.set((unsigned char) 'E');
797  if (icase)
798  letter.set((unsigned char) 'e');
799  }
800  else literal = false;
801  ++first;
802  return -1;
803  }
804 
805  if (literal) {
806  letter.set((unsigned char) '\\');
807  return -1;
808  }
809 
810  if (*first == 'x') { // hexadecimal character \x01 through \xFF
811  if (++first == last)
812  return token_error(first - begin, "\\x used with no following hex digits");
813  int n = 0;
814  if (*first >= '0' && *first <= '9') {
815  n = 16 * (*first - '0');
816  }
817  else
818  if (*first >= 'A' && *first <= 'F') {
819  n = 16 * (10 + *first - 'A');
820  }
821  else
822  if (*first >= 'a' && *first <= 'f') {
823  n = 16 * (10 + *first - 'a');
824  }
825  else return token_error(first - begin, "\\x used with no following hex digits");
826 
827  if (++first == last)
828  return token_error(first - begin, "\\x used with only one following hex digit");
829 
830  if (*first >= '0' && *first <= '9') {
831  n += *first - '0';
832  }
833  else
834  if (*first >= 'A' && *first <= 'F') {
835  n += 10 + (*first - 'A');
836  }
837  else
838  if (*first >= 'a' && *first <= 'f') {
839  n += 10 + (*first - 'a');
840  }
841  else return token_error(first - begin, "\\x used with only one following hex digit");
842  letter.set(n);
843  ++first;
844  return -1;
845  }
846 
847  if (*first == 'c') { // control character \cA through \cZ maps to \x01 \x1A
848  if (++first == last || *first < 'A' || *first > 'Z')
849  return token_error(first - begin, "\\c used with no following upper-case letter");
850  letter.set(*first - 'A' + 1);
851  ++first;
852  return -1;
853  }
854 
855  // look for \s, \S, \w, \W, \d, \D, \n, \t, \v, \V, \r, \a, \e, \f, \h, \H
856  std::map<string, bitset<256> >::const_iterator c = char_sets.find(std::string("\\") + *first);
857  if (c != char_sets.end()) {
858  letter = c->second;
859  ++first;
860  return -1;
861  }
862 
863  if (*first == '.') {
864  letter.set((unsigned char) '.');
865  ++first;
866  return -1;
867  }
868  }
869 
870  if (*first == '.') {
871  if (literal)
872  letter.set((unsigned char) '.');
873  else {
874  letter.set();
875 
876 
877 
878 
879 
880  // letter.set(0, false);
881 
882 
883 
884 
885 
886  }
887  }
888  else
889  if (icase && isalpha(*first)) {
890  letter.set((unsigned char) tolower((unsigned char) *first));
891  letter.set((unsigned char) toupper((unsigned char) *first));
892  }
893  else letter.set((unsigned char) *first);
894  ++first;
895  return -1;
896  }

898  // Return error position or -1 if ok
899  template <typename OutputIterator>
900  int non_deterministic_regexp_capture_cursor::tokenize(const char *first, const char *last, OutputIterator x)
901  {
902  static map<string, bitset<256> > char_sets;
903  if (char_sets.empty())
904  build_char_sets(char_sets);
905 
906  *x++ = node(OPEN);
907  node previous = node(OPEN);
908 
909  bool literal = false;
910  std::vector<char> parenthesis; // contains '(' or 'i' for (?i:
911  bool icase = false;
912  std::vector<int> open_captures;
913  int current_open_capture = 0;
914 
915  for(const char *begin = first; first != last; ++first, ++x) {
916  if (literal && *first != '\\') {
917  if (previous.type != OPEN && previous.type != OR)
918  *x++ = node(CONCAT);
919  if (icase && isalpha(*first)) {
920  bitset<256> c;
921  c.set((unsigned char) tolower(*first));
922  c.set((unsigned char) toupper(*first));
923  *x = previous = node(c);
924  }
925  else
926  *x = previous = node(*first);
927  continue;
928  }
929 
930  switch (*first) {
931  case '|' :
932  if (first == begin)
933  return token_error(0, "missing left part of alternative");
934  *x = previous = node(OR);
935  break;
936 
937  case '(' :
938  if (previous.type != OPEN && previous.type != OR)
939  *x++ = node(CONCAT);
940  if (first + 2 < last && first[1] == '?' && first[2] == ':') { // non capturing parenthesis (?:
941  first += 2;
942  parenthesis.push_back('(');
943  }
944  else {
945  if (first + 3 < last && first[1] == '?' &&
946  first[2] == 'i' && first[3] == ':') { // case insensitive (?i:
947  first += 3;
948  if (icase == false) {
949  parenthesis.push_back('i');
950  icase = true;
951  }
952  else parenthesis.push_back('(');
953  }
954  else { // capture
955  ++capture_count_;
956  *x++ = node(OPEN);
957  *x++ = node(OPENCAPTURE, current_open_capture, current_open_capture / 2);
958  open_captures.push_back(current_open_capture);
959  current_open_capture += 2;
960  *x++ = node(CONCAT);
961  parenthesis.push_back('c');
962  }
963  }
964  *x = previous = node(OPEN);
965  break;
966 
967  case ')' :
968  if (parenthesis.empty())
969  return token_error(first - begin, "unbalanced parenthesis");
970  if (parenthesis.back() == 'i')
971  icase = false;
972  else
973  if (parenthesis.back() == 'c') { // end of capture ?
974  ++capture_count_;
975  *x++ = node(CLOSE);
976  *x++ = node(CONCAT);
977  *x++ = node(CLOSECAPTURE, open_captures.back() + 1, open_captures.back() / 2);
978  open_captures.pop_back();
979  }
980  *x = previous = node(CLOSE);
981  parenthesis.pop_back();
982  break;
983 
984  case '*' :
985  if (first == begin)
986  return token_error(0, "missing expression before operator *");
987  *x = previous = node(STAR);
988  break;
989 
990  case '?' :
991  if (first == begin)
992  return token_error(0, "missing expression before operator ?");
993  *x = previous = node(QUESTION);
994  break;
995 
996  case '+' :
997  if (first == begin)
998  return token_error(0, "missing expression before operator +");
999  *x = previous = node(PLUS);
1000  break;
1001 
1002  case '{' : // {n,m}
1003  if (first == begin)
1004  return token_error(0, "missing expression before repeat operator");
1005  if (++first == last)
1006  return token_error(first - begin, "unterminated expression");
1007  {
1008  int n = 0;
1009  const char *tmp = first;
1010  while (isdigit(*first)) {
1011  n = n * 10 + *first - '0';
1012  if (++first == last)
1013  return token_error(first - begin, "unterminated repeat operator");
1014  }
1015  int m = n;
1016  if (*first == ',') {
1017  if (++first == last)
1018  return token_error(first - begin, "unterminated repeat operator");
1019  tmp = first;
1020  for(m = 0; isdigit(*first);) {
1021  m = m * 10 + *first - '0';
1022  if (++first == last)
1023  return token_error(first - begin, "unterminated repeat operator");
1024  }
1025  if (tmp == first) m = -1; // comma but no upper bound: {n,}
1026  }
1027  if (*first != '}')
1028  return token_error(first - begin, "invalid repeat operator");
1029 
1030  // errors: {0} {0,0} {,0}
1031  if (n > 32768 || m > 65535 || (n > m && m > -1) || (n == m && n == 0))
1032  return token_error(first - begin, "wrong range for repeat operator");
1033 
1034  // special cases: {0,1} {,1} {0,} {1,} {1,1}
1035  if (n == 0 && m == 1)
1036  *x = node(QUESTION);
1037  else if (n == 0 && m == -1)
1038  *x = node(STAR);
1039  else if (n == 1 && m == -1)
1040  *x = node(PLUS);
1041  else if (!(n == 1 && m == 1))
1042  *x = node(REPEAT, n, m);
1043  }
1044  break;
1045 
1046  case '[' : // characters set
1047  {
1048  if (previous.type != OPEN && previous.type != OPENCAPTURE && previous.type != OR)
1049  *x++ = node(CONCAT);
1050  bitset<256> a;
1051  bool v = true;
1052  if (++first == last)
1053  return token_error(first - begin, "unterminated expression");
1054  if (*first == '^') { // a negation ?
1055  a.set();
1056  a.set(0, false);
1057  v = false;
1058  if (++first == last)
1059  return token_error(first - begin, "unterminated expression");
1060  }
1061  if (*first == '-') { // "escaped" hyphen at the beginning ?
1062  a.set((unsigned char) '-', v);
1063  ++first;
1064  }
1065  else
1066  if (*first == ']') { // "escaped" closing square bracket at the beginning ?
1067  a.set((unsigned char) ']', v);
1068  ++first;
1069  }
1070  for(; first != last && *first != ']'; ++first) {
1071  if (*first == '[') {
1072  if (++first == last) break;
1073  if (*first != ':') {
1074  a.set((unsigned char) '[', v);
1075  }
1076  else { // a predefined set as [:digits:]
1077  const char *start = first - 1;
1078  first = std::find(++first, last, ']');
1079  if (first == last) break;
1080  std::map<string, bitset<256> >::const_iterator c = char_sets.find(std::string(start, first + 1));
1081  if (c == char_sets.end())
1082  return token_error(start - begin, "unknown character set");
1083  if (v == true)
1084  a |= c->second;
1085  else a &= ~c->second;
1086  continue;
1087  }
1088  }
1089  // TODO: next_char manages ranges and return a character set (pass a boolean for
1090  // saying this is a character set context and '-' as a special meaning, good for
1091  // . also, see below the ugly hack)
1092  bitset<256> tmp; // will possibly be used as a range lower bound
1093  if (*first != '-') { // not a potential character range ?
1094  switch (*first) {
1095  case '^' :
1096  case '$' :
1097  tmp.set((unsigned char) *first++);
1098  break;
1099  default :
1100  {
1101  int r = next_char(tmp, begin, first, last, literal, char_sets, icase);
1102  if (r != -1)
1103  return r;
1104  // ugly hack for dot
1105  if (tmp.count() == 256) {
1106  tmp.reset();
1107  tmp.set((unsigned char) '.'); // . is a literal in a character set
1108  }
1109  }
1110  }
1111  if (v == true)
1112  a |= tmp;
1113  else a &= ~ tmp; // negative definition
1114  if (*first != '-') {
1115  --first;
1116  continue;
1117  }
1118  }
1119  // now *first == '-'
1120  if (++first == last)
1121  return token_error(first - begin, "unterminated expression");
1122  if (*first == ']') { // unterminated character range ?
1123  a.set((unsigned char) '-', v);
1124  break; // end of character set
1125  }
1126  else { // character range [a-z]
1127  if (tmp.count() != 1) { // a bound must not be a character set
1128  return token_error(first - 2 - begin, "wrong character range: the lower bound cannot be a character set");
1129  }
1130  bitset<256> upper_bound;
1131  switch (*first) {
1132  case '^' :
1133  case '$' :
1134  upper_bound.set((unsigned char) *first);
1135  break;
1136  default :
1137  {
1138  int r = next_char(upper_bound, begin, first, last, literal, char_sets, icase);
1139  --first;
1140  if (r != -1)
1141  return token_error(first - begin, "wrong character range");
1142  // ugly hack for dot
1143  if (upper_bound.count() == 256) {
1144  upper_bound.reset();
1145  upper_bound.set((unsigned char) '.'); // . is a literal in a character set
1146  }
1147  }
1148  }
1149  if (upper_bound.count() != 1) // a bound must not be a character set
1150  return token_error(first - begin, "wrong character range: the upper bound cannot be a character set");
1151 
1152  unsigned int int_lower_bound = 0, int_upper_bound = 0;
1153  while (tmp.test(int_lower_bound) == false)
1154  ++int_lower_bound;
1155  while (upper_bound.test(int_upper_bound) == false)
1156  ++int_upper_bound;
1157 
1158  // guess what, lower bound should be lower or equal to upper bound
1159  if (int_upper_bound < int_lower_bound)
1160  return token_error(first - begin, "wrong character range: lower bound cannot be greater than upper bound");
1161  else
1162  for(unsigned int i = int_lower_bound; i <= int_upper_bound; ++i)
1163  a.set(i, v);
1164  }
1165  }
1166  // first is supposed to point to ']'
1167  if (first == last)
1168  return token_error(first - begin, "unterminated expression");
1169  *x = previous = node(a);
1170  break;
1171  }
1172 
1173  default :
1174  bitset<256> next_letter;
1175  int r = next_char(next_letter, begin, first, last, literal, char_sets, icase);
1176  if (r != -1)
1177  return r;
1178  if (next_letter.any()) {
1179  if (previous.type != OPEN && previous.type != OR)
1180  *x++ = node(CONCAT);
1181 
1182 
1183 
1184  if (next_letter.count() == 256) // catch-all dot
1185 
1186 
1187 
1188 #ifndef UTF8
1189  *x = previous = node(ANY);
1190 #else
1191  {
1192  *x++ = node(OPEN);
1193  next_letter.reset();
1194  for(int i = 1; i < 0x7F; ++i)
1195  next_letter.set(i);
1196  *x++ = node(next_letter);
1197  *x++ = node(OR);
1198  next_letter.reset();
1199  for(int i = 0; i < 32; ++i)
1200  next_letter.set(i | 192);
1201  *x++ = node(next_letter);
1202  *x++ = node(CONCAT);
1203  *x++ = node(ANY);
1204  *x = previous = node(CLOSE);
1205  }
1206 #endif
1207  else
1208  *x = previous = node(next_letter);
1209  }
1210  --first;
1211  break;
1212  }
1213  }
1214  *x++ = node(CLOSE);
1215  *x++ = node(CONCAT);
1216  *x++ = node(FINAL);
1217  *x++ = node(CONCAT);
1218  *x++ = node(GRANDFINAL);
1219  return -1;
1220  }

1279  non_deterministic_regexp_capture_cursor::node*
