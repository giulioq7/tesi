76 template <typename FA>
77 typename FA::state_type build(FA &A, istream &in)
78 {
79  map<int, safe<typename FA::state_type, FA::null_state> > m;
80  typename FA::state_type i = FA::null_state;
81  int q, p;
82  typename FA::char_type c;
83 
84  if (next_transition(in, q, c, p) != -1) {
85  m[q] = A.new_state();
86  A.final(m[q]) = q < 0;
87  safe<typename FA::state_type, FA::null_state> &pp = m[p];
88  if (pp == FA::null_state) {
89  pp = A.new_state();
90  A.final(pp) = p < 0;
91  }
92  A.set_trans(m[q], c, pp);
93  i = m[q];
94  }
95 
96  while (next_transition(in, q, c, p) != -1) {
97  safe<typename FA::state_type, FA::null_state> &qq = m[q];
98  safe<typename FA::state_type, FA::null_state> &pp = m[p];
99  if (qq == FA::null_state) {
100  qq = A.new_state();
101  A.final(qq) = q < 0;
102  }
103  if (pp == FA::null_state) {
104  pp = A.new_state();
105  A.final(pp) = p < 0;
106  }
107  A.set_trans(qq, c, pp);
108  }
109 
110  return i;
111 }
