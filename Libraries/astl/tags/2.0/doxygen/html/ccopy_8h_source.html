59 template <typename DFirstC, typename DFA>
60 typename DFA::state_type
61 trim(DFA &out, DFirstC first, DFirstC last = DFirstC())
62 {
63  typedef map<typename DFirstC::state_type,
64  typename DFA::state_type> mapper;
65  mapper mapping;
66  typename DFirstC::state_type i = first.src();
67 
68  while (first != last) { // while not end of range
69  while(first.forward()); // get down as much as possible
70 
71  // copy aim state if final:
72  typename DFA::state_type p;
73  if (first.aim_final()) {
74  typename mapper::iterator i =
75  mapping.insert(make_pair(first.aim(), DFA::null_state)).first;
76  if (i->second == DFA::null_state) { // inserted ?
77  i->second = out.new_state();
78  out.tag(i->second) = first.aim_tag(); // don't reorder (dot)
79  out.final(i->second) = true;
80  }
81  }
82 
83  do { // while moving backward (pop) copy current
84  // transition (q, first.letter(), p)
85  // copy source state if needed:
86  typename mapper::const_iterator dest = mapping.find(first.aim());
87  if (dest != mapping.end()) p = dest->second;
88  else p = DFA::null_state;
89 
90  if (p != DFA::null_state || first.src_final()) {
91  typename mapper::iterator i = mapping.insert(make_pair(first.src(), DFA::null_state)).first;
92  if (i->second == DFA::null_state) {
93  i->second = out.new_state();
94  out.tag(i->second) = first.src_tag(); // don't reorder (dot)
95  out.final(i->second) = first.src_final();
96  }
97  if (p != DFA::null_state) // copy transition ?
98  out.set_trans(i->second, first.letter(), p);
99 
100  // shift (q, first.letter(), p) => (q', first.letter()', q)
101  p = i->second;
102  }
103  } while (! first.forward());
104  }
105  return mapping[i];
106 }
