48  template <typename Sigma, typename State>
49  class matrix_line
50  {
51  public:
52  unsigned int size_;// a count of the outgoing transitions
53 #ifdef MATRIX_PATCH
54  State *v;
55 #else
56  State v[Sigma::size];
57 #endif
58 
59  typedef matrix_line self;
60 
61  matrix_line() : size_(0) {
62 #ifdef MATRIX_PATCH
63  v = new State[Sigma::size];
64 #endif
65  memset(v, 0, Sigma::size * sizeof(State));
66 
67  }
68 
69 #ifdef MATRIX_PATCH
70  ~matrix_line() {
71  delete [] v;
72  }
73 #endif
74 
75  matrix_line(const self &x)
76  : size_(x.size_) {
77 #ifdef MATRIX_PATCH
78  v = new State[Sigma::size];
79 #endif
80  memcpy(v, x.v, Sigma::size * sizeof(State));
81  }
82 
83  self& operator=(const self &x) {
84  size_ = x.size_;
85  memcpy(v, x.v, Sigma::size * sizeof(State));
86  return (*this);
87  }
88 
89  bool operator==(const self &x) const {
90  return size_ == x.size_ && equal(v, v + Sigma::size, x.v);
91  }
92 
93  size_t size() const {
94  return size_;
95  }
96 
97  unsigned int& size() {
98  return size_;
99  }
100  };
