491 template <typename StackCursor, typename MarkerFunction = none>
492 class dfirst_cursor : public dfirst_cursor_concept
493 {
494 protected:
495  StackCursor c;
496  bool has_poped;
497  MarkerFunction visited;
498 
499 public:
500  typedef dfirst_cursor self;
501  typedef dfirst_cursor_concept super;
502  typedef typename StackCursor::state_type state_type;
503  typedef typename StackCursor::char_type char_type;
504  typedef typename StackCursor::tag_type tag_type;
505  typedef typename StackCursor::char_traits char_traits;
506  typedef typename super::concept concept;
507 
509  dfirst_cursor(const StackCursor &x,
510  const MarkerFunction &f = MarkerFunction())
511  : c(x), has_poped(false), visited(f) {
512  visited(c.src()); // set c source to 'visited'
513  }
514 
517  dfirst_cursor()
518  : c(), has_poped(false)
519  { }
520 
524  bool forward() {
525  if (has_poped) {
526  // move on to the next outgoing transition if any otherwise pop
527  if (!c.next())
528  return !c.backward(); // at edges end
529  else {
530  has_poped = false;
531  return true;
532  }
533  }
534 
535  c.forward();
536  if (visited(c.src()) || !c.first()) {
537  c.backward(); // cannot forward so backward => pop
538  has_poped = true;
539  return false;
540  }
541  return true;
542  }
543 
546  void backward() { has_poped = true; } // force pop
547 
550  char_type letter() const { return c.letter(); }
551 
555  bool aim_final() const { return c.aim_final(); }
556 
559  bool src_final() const { return c.src_final(); }
560 
563  state_type src() const { return c.src(); }
564 
567  state_type aim() const { return c.aim(); }
568 
570  bool operator==(const self &x) const { return c == x.c; }
571 
573  bool operator!=(const self &x) const { return !(*this == x); }
574 
577  tag_type src_tag() const { return c.src_tag(); }
578 
581  tag_type aim_tag() const { return c.aim_tag(); }
582 
584  const StackCursor& stack() const { return c; }
585 };

733 // Builds a depth first cursor from a DFA pointing to the first
734 // outgoing transition of the initial state if any by constructing the
735 // underlying forward and stack cursors. If the DFA has no initial
736 // state or if it has no outgoing transitions, returns an empty cursor
737 template <typename DFA>
738 typename dfirst_cursor_type<DFA, DFA_concept>::model
739 dfirstc_(const DFA &x, DFA_concept)

743  return dfirst_cursor<stack_cursor<forward_cursor<DFA> > >(stackc(fc));
744  else
745  return dfirst_cursor<stack_cursor<forward_cursor<DFA> > >();
