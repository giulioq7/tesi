65 // Computes the height of each state in A and
66 // stores it in a matrix (height x state)
67 // yields a partition of Q
68 // if q is in Q, height(q) = Max {height(p) such that (q, a, p) is in T}
69 // and height(q) = 0 if q has no outgoing edges
70 // Uses method Tag::degree_in(long) to store the number of ingoing edges of each state
71 // Needs template functions swap and max
72 
73 template <class DFA, class Dstate>
74 unsigned char
75 height(DFA& A, Dstate q, vector<vector < Dstate> *> &partition)
76 {
77  const typename DFA::Edges edg = A.delta2(q);
78  unsigned char max_height = 0;
79  typename DFA::Edges::const_iterator q_trans, edg_end = edg.end();
80  Dstate aim;
81 
82  for (q_trans = edg.begin(); q_trans != edg_end; ++q_trans)
83  {
84  aim = (*q_trans).second;
85  A.tag(aim).degree_in(A.tag(aim).degree_in() + 1);
86  if (A.tag(aim).degree_in()>1) //la hauteur est deja calculée
87  max_height = max__(max_height,(unsigned char)(A.tag(aim).height()+1));
88  else
89  max_height = max__(max_height,(unsigned char)(height(A,aim,partition)+1));
90  }
91  if (partition.size() <= max_height)
92  partition.push_back(new vector<Dstate>(1, q));
93  else
94  partition[max_height]->push_back(q);
95  A.tag(q).height(max_height);
96  return (max_height);
97 }

102 template <class DFA, class iterator>
103 void
104 fusion(DFA &A, iterator start, iterator finish)
105 {
106  typedef typename DFA::state_type Dstate;
107  typename vector<Dstate>::iterator where;
108  Dstate reference = *start;
109 
110  for (where = start + 1; where != finish; ++where)
111  A.tag(*where).equivalent_to((unsigned long) reference);
112 }

114 // Redirect the outgoing edges of a partition based on height
115 // to the partition reference state and decrease the number of
116 // ingoing edges of aim. (delete aim if it has no ingoing edges any more)
117 template <class DFA, class State>
118 void
119 redirect_edges(DFA &A, vector<State> &v)
120 {
121  typename vector<State>::iterator where;
122  for (where = v.begin(); where != v.end(); ++where)
123  {
124  const typename DFA::Edges edg = A.delta2(*where);
125  typename DFA::Edges::const_iterator trans, edg_end = edg.end();
126  typename DFA::state_type new_aim, aim;
127  for (trans = edg.begin(); trans != edg_end; ++trans)
128  {
129  aim = (*trans).second;
130  new_aim = (typename DFA::state_type) A.tag(aim).equivalent_to();
131  if (new_aim != A.null_state)
132  {
133  A.change_trans(*where, (*trans).first, new_aim);
134  A.tag(aim).degree_in(A.tag(aim).degree_in() - 1);
135  if (A.tag(aim).degree_in() == 0) // no more ingoing edges to the aim
136  A.del_state(aim);
137  }
138  }
139  }
140 }

142 // Compares two states and returns int <0 or ==0 or >0
143 template <class DFA, class State>
144 int
145 compare_states(DFA &A, State q, State p)
146 {
147  bool fq = A.final(q);
148  bool fp = A.final(p);
149  if (fq != fp)
150  return ((int) fq - (int) fp);
151 
152  typename DFA::Edges::size_type q_trans_count = A.delta2(q).size();
153  typename DFA::Edges::size_type p_trans_count = A.delta2(p).size();
154  if (q_trans_count - p_trans_count != 0)
155  return (q_trans_count - p_trans_count);
156  if (q_trans_count == 0) return (0);
157 
158  // Both states have the same finality and the same number of transitions
159  // We apply a kind of lexicographical comparison
160  typename DFA::Edges::const_iterator left_edg_end = A.delta2(q).end();
161  typename DFA::Edges::const_iterator left = A.delta2(q).begin();
162  typename DFA::Edges::const_iterator right = A.delta2(p).begin();
163 
164  for (; left != left_edg_end; ++left, ++right)
165  {
166  typedef typename DFA::char_type Dalph;
167  typedef typename DFA::state_type Dstate;
168  const pair<const Dalph, Dstate>& q_trans = *left;
169  const pair<const Dalph, Dstate>& p_trans = *right;
170 
171  if (q_trans.first == p_trans.first) // Same letter
172  {
173  if (q_trans.second < p_trans.second)
174  return (-1);
175  else
176  if (p_trans.second < q_trans.second)
177  return (1);
178  }
179  else
180  if (q_trans.first < p_trans.first)
181  return (-1);
182  else
183  return (1);
184  }
185  return (0);
186 }

189 // Groups equivalent states and returns the range they are in
190 template <class DFA, class Size_type, class State>
191 void
192 group_equivalent_states(DFA &A, vector<State> &v, Size_type start,
193  Size_type finish, pair<Size_type, Size_type> &rang)
194 {
195  // Result is placed in 'range' argument
196  Size_type g, d, pg;
197  pg = g = start + 1;
198  d = finish;
199  State e = v[start]; // Reference state
200  int res;
201 
202  while (g <= d) {
203  res = compare_states(A, v[g], e);
204  if (res == 0) {
205  swap(v[pg],v[g]);
206  pg++;
207  g++;
208  }
209  else
210  if (res < 0) g++;
211  else {
212  res = compare_states(A, v[d], e);
213  if (res == 0)
214  if(g == pg) {
215  swap(v[pg],v[d]);
216  pg++;
217  g++;
218  }
219  else {
220  swap(v[pg],v[g]);
221  swap(v[pg],v[d]);
222  pg++;
223  }
224  else
225  if(res > 0) d--;
226  else {
227  swap(v[g],v[d]);
228  g++;
229  }
230  }
231  }
232 
233  rang.second = d;
234  rang.first = pg;
235 }

237 // Refine the partition based on state height
238 // All equivalent states are placed together in the vector then merged
239 template <class DFA, class Size_type, class State>
240 void
241 refine_partition(DFA &A, vector<State> &v,
242  Size_type start, Size_type finish)
243 {
244  if (start < finish)
245  {
246  pair<typename vector<State>::size_type, typename vector<State>::size_type> range;
247 
248  group_equivalent_states(A, v, start, finish, range);
249  if (start < range.first - 1) fusion(A, v.begin() + start, v.begin() + range.first);
250  if (range.first < range.second) refine_partition(A, v, range.first, range.second);
251  if (range.second + 1 < finish) refine_partition(A, v, range.second + 1, finish);
252  }
253 }

256 // Main minimization function
257 template <class DFA>
258 void
259 acyclic_minimization(DFA &dfa)
260 {
261  if (dfa.state_count() > 0) {
262  typedef typename DFA::state_type Dstate;
263  typedef vector<vector < Dstate> *> matrix;
264  matrix partition(1, new vector< Dstate>);
265  height(dfa,dfa.initial(),partition);
266 
267  if (partition[0]->empty() == false) {
268  typename matrix::iterator where;
269  for (where = partition.begin(); where != partition.end(); ++where) {
270  redirect_edges(dfa, **where);
271  refine_partition(dfa, **where, (typename matrix::size_type) 0,
272  (**where).size() - 1);
273  delete (*where);
274  }
275  }
276  }

279 template <typename DFA, typename NFA>
280 void
281 reverse(const DFA &a1, NFA &a2)
282 {
283  map<typename DFA::state_type,
284  safe<typename NFA::state_type, NFA::null_state> > m;
285 
286  // for each state:
287  for(typename DFA::const_iterator src = a1.begin();
288  src != a1.end(); ++src) {
289  // create a new state if needed:
290  safe<typename NFA::state_type, NFA::null_state> &new_aim = m[*src];
291  if (new_aim == NFA::null_state) {
292  new_aim = a2.new_state();
293  a2.final(new_aim) = (a1.initial() == *src);
294  if (a1.final(*src))
295  a2.initial().insert(new_aim);
296  }
297 
298  // for each transition:
299  typename DFA::edges_type t = a1.delta2(*src);
300  for(typename DFA::edges_type::const_iterator i = t.begin();
301  i != t.end(); ++i) {
302  // create a new state if needed:
303  safe<typename NFA::state_type, NFA::null_state> &new_src = m[(*i).second];
304  if (new_src == NFA::null_state) {
305  new_src = a2.new_state();
306  a2.final(new_src) = a1.initial() == new_src;
307  if (a1.final((*i).second))
308  a2.initial().insert(new_src);
309  }
310  a2.set_trans(new_src, (*i).first, new_aim);
311  }
312  }
313 }

315 // Requirements: DFA1::char_type convertible to DFA2::char_type
316 template <typename DFA1, typename DFA2>
317 void
318 brzozowski(const DFA1 &a1, DFA2 &a2)
319 {
320  typedef NFA_mmap<typename DFA1::char_traits> nfa_type;
321 
322  if (a1.state_count() > 0) {
323  nfa_type tmp2;
324  {
325  DFA1 tmp3;
326  {
327  nfa_type tmp1;
328  reverse(a1, tmp1);
329  tmp3.initial(clone(tmp3, dfirst_markc(forwarddc(tmp1))));
330  }
331  reverse(tmp3, tmp2);
332  }
333  a2.initial(clone(a2, dfirst_markc(forwarddc(tmp2))));
334  }
