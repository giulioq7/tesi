136  void del_state(state_type s)
137  {
138  if (s == initial()) initial(null_state);
139  trans_count_ -= Q[s]->edges().size();
140  delete Q[s];
141  Q[s] = NULL;
142  --state_count_;
143  }
144 
151  void copy_state(state_type from, state_type to) {
152  trans_count_ += Q[from]->edges().size() - Q[to]->edges().size();
153  *Q[to] = *Q[from];
154  F[to] = F[from];
155  }

161  state_type duplicate_state(state_type q)
162  {
163  Q.push_back(new StateData(*Q[q]));
164  F.push_back(F[q]);
165  ++state_count_;
166  trans_count_ += Q[q]->edges().size();
167  return Q.size() - 1;
168  }

194  DFA_base(unsigned long n) :
195  Q(1, (StateData*) 0), I(0), F(1, '\0'), state_count_(0UL),

200  ~DFA_base()
