45 template <class CharTraits = plain>
46 class DFA_hash : public DFA_concept
47 {
48 public:
49  typedef CharTraits char_traits;
50  typedef typename char_traits::char_type char_type;
51  typedef empty_tag tag_type;
52  typedef size_t state_type;
53 
54 protected:
55  typedef pair<state_type, char_type> key;
56  typedef state_type data;
57  struct hash_function
58  : public unary_function<pair<state_type, char_type>, size_t>
59  {
60  size_t operator() (const pair<state_type, char_type> &x) const {
61  return x.first * 1023 + char_traits::to_int_type(x.second);
62  }
63  };
64 
65  typedef HASH_TABLE_TYPE<key, data,
66  hash_function, equal_to<key> > hasher_type;
67  hasher_type hasher;
68 
69  unsigned long state_count_;
70  state_type current;
71  state_type I;
72  vector<char> F;
73  tag_type bogus;
74 
75 public:
76  static const state_type null_state = 0;
77 
78  DFA_hash(unsigned long = 0)
79  : state_count_(0), current(0),
80  I(0), F(1, '\0')
81  { }
82 
83  state_type new_state() {
84  ++state_count_;
85  F.push_back('\0');
86  return ++current;
87  }
88 
89  template <class OutputIterator>
90  OutputIterator new_state(unsigned long how_many, OutputIterator x) {
91  for(; how_many > 0; --how_many)
92  *x++ = new_state();
93  return x;
94  }
95 
96  bool final(state_type s) const {
97  return F[s] != '\0';
98  }
99 
100  char& final(state_type s) {
101  return F[s];
102  }
103 
104  state_type initial() const {
105  return (I);
106  }
107 
108  void initial(state_type s) {
109  I = s;
110  }
111 
112  void set_trans(state_type s, const char_type &a, state_type aim) {
113  hasher[make_pair(s, a)] = aim;
114  }
115 
116  void del_trans(state_type s, const char_type &a) {
117  hasher.erase(make_pair(s, a));
118  }
119 
120  void change_trans(state_type s, const char_type &a, state_type new_aim) {
121  hasher[make_pair(s, a)] = new_aim;
122  }
123 
124  state_type delta1(state_type s, const char_type &a) const {
125  typename hasher_type::const_iterator i = hasher.find(make_pair(s, a));
126  return i == hasher.end() ? null_state : i->second;
127  }
128 
129  unsigned long state_count() const {
130  return (state_count_);
131  }
132 
133  unsigned long trans_count() const {
134  return hasher.size();
135  }
136 
137  tag_type& tag(state_type) {
138  return bogus;
139  }
140 
141  const tag_type& tag(state_type) const {
142  return bogus;
143  }
144 };
